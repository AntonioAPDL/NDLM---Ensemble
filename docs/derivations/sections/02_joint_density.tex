\section{Full Joint Density and Posterior Factorization}

\subsection{Model A Joint}
Define $\Theta_A=(\sigma_0^2)$, with $\lambda$ and the historical covariance sequence
$\{\tilde{\bm W}_t\}_{t=1}^T$ treated as fixed known inputs.
The complete joint for historical target-only data is
\begin{align}
p(\bm y_{1:T}^0,\bm\alpha_{0:T},\Theta_A)
&= p(\bm\alpha_0)\,p(\Theta_A)
\prod_{t=1}^T p(\bm\alpha_t\given\bm\alpha_{t-1},\tilde{\bm W}_t,\lambda)
\prod_{t=1}^T p(y_t^0\given\bm\alpha_t,\sigma_0^2).\label{eq:joint_A}
\end{align}

\subsection{Model B Joint}
Define
\begin{align}
\Theta_B=\Theta_A\cup\{\sigma_1^2,\dots,\sigma_J^2\}.
\end{align}
Then
\begin{align}
&p(\bm y_{1:T}^0,\bm z_{1:T}^{1:J},\bm\alpha_{0:T},\bm\delta_{0:T}^{1:J},\Theta_B) \\
&= p(\bm\alpha_0)\prod_{j=1}^J p(\bm\delta_0^j)\,p(\Theta_B)
\prod_{t=1}^T p(\bm\alpha_t\given\bm\alpha_{t-1},\tilde{\bm W}_t,\lambda)
\prod_{t=1}^T p(y_t^0\given\bm\alpha_t,\sigma_0^2)
\prod_{j=1}^J\prod_{t=1}^T p(\bm\delta_t^j\given\bm\delta_{t-1}^j,\bm W_t^{\delta^j}) \\
&\hspace{3em}\times
\prod_{j=1}^J\prod_{t=1}^T p(z_t^j\given\bm\alpha_t,\bm\delta_t^j,\sigma_j^2).\label{eq:joint_B}
\end{align}

\subsection{Model C Joint}
Choose either the reduced or augmented forecast-state specification from Section~1 and,
in the augmented case, condition additionally on the supplied covariate path
$\bm c_{T+1:T+K_{\max}}^\star$.
Let
\begin{align}
\Phi_C=\{\bm W_{T+k}^{(f)}\}_{k=1}^{K_{\max}}.
\end{align}
For forecast observations $\mathcal{Y}_{T+1:T+K_{\max}}$, conditional on the forecast initial state
$\bm x_T^{(f)}$ and on the relevant observation variances $\sigma_1^2,\dots,\sigma_{J_f}^2$,
\begin{align}
p(\mathcal{Y}_{T+1:T+K_{\max}},\bm x_{T+1:T+K_{\max}}^{(f)},\Phi_C\given \bm x_T^{(f)},\sigma_{1:J_f}^2)
&= p(\Phi_C)
\prod_{k=1}^{K_{\max}} p(\bm x_{T+k}^{(f)}\given\bm x_{T+k-1}^{(f)},\bm W_{T+k}^{(f)}) \nonumber\\
&\qquad\times
\prod_{k=1}^{K_{\max}}\prod_{n=1}^{n_k} p(y_{T+k,n}\given\bm x_{T+k}^{(f)},\sigma_{\mathrm{src}(k,n)}^2).
\label{eq:joint_C}
\end{align}
The prior factor $p(\Phi_C)$ contains independent inverse-Wishart priors for
each $\bm W_{T+k}^{(f)}$; there is no pooled shared forecast covariance.
If Model C is run as a standalone conditional block, prepend the external Gaussian anchor
$p(\bm x_T^{(f)})=\N(\bm m_{f0},\bm C_{f0})$ and the observation-variance prior
$\prod_{j=1}^{J_f}p(\sigma_j^2)$.

\subsection{Unified A+B+C Joint}
Let $\mathcal{Y}^{(f)}:=\mathcal{Y}_{T+1:T+K_{\max}}$ and let the forecast bridge
$\bm x_T^{(f)}=\bm H_T^{(f)}(\bm\alpha_T,\bm\delta_T^{1:J_f})$ be given by
\eqref{eq:fcast_bridge}. Then the unified Gaussian NDLM joint density is
\begin{align}
&p(\bm y_{1:T}^0,\bm z_{1:T}^{1:J},\mathcal{Y}^{(f)},
\bm\alpha_{0:T},\bm\delta_{0:T}^{1:J},
\bm x_{T:T+K_{\max}}^{(f)},\Theta_B,\Phi_C) \nonumber\\
&\quad=
p(\bm y_{1:T}^0,\bm z_{1:T}^{1:J},\bm\alpha_{0:T},\bm\delta_{0:T}^{1:J},\Theta_B)\,
\delta_{\bm H_T^{(f)}(\bm\alpha_T,\bm\delta_T^{1:J_f})}\!\big(\bm x_T^{(f)}\big) \nonumber\\
&\qquad\qquad\times
p(\mathcal{Y}^{(f)},\bm x_{T+1:T+K_{\max}}^{(f)},\Phi_C\given \bm x_T^{(f)},\sigma_{1:J_f}^2),
\label{eq:joint_ABC}
\end{align}
where $\delta_{\bm a}(\cdot)$ denotes a point-mass at $\bm a$.
Equation \eqref{eq:joint_ABC} is the single coherent probabilistic model underlying the separate
block derivations that follow.

\subsection{Posterior Kernel}
The posterior for any model block $\mathcal{M}\in\{A,B,C,ABC\}$ is
\begin{align}
p(\text{states},\Theta\given\text{data})\propto
p(\text{data},\text{states},\Theta),
\label{eq:posterior_kernel}
\end{align}
with the explicit factorization given by \eqref{eq:joint_A}--\eqref{eq:joint_ABC}.
These product forms are the basis for Gibbs blocks and CAVI factors.

\subsection{Computable complete log-joint forms}
For implementation, it is convenient to evaluate each model with explicit residual and innovation sums.
Define
\begin{align}
\bm e_t^{A}&:=\bm\alpha_t-\tilde{\bm G}_t\bm\alpha_{t-1},
& r_t^{0}&:=y_t^0-\tilde{\bm F}_t^\T\bm\alpha_t, \nonumber\\
\bm e_t^{\delta,j}&:=\bm\delta_t^j-\bm G_t\bm\delta_{t-1}^j,
& r_t^{j}&:=z_t^j-\tilde{\bm F}_t^\T\bm\alpha_t-\bm F_t^\T\bm\delta_t^j,\nonumber\\
\bm e_{T+k}^{(f)}&:=\bm x_{T+k}^{(f)}-\bm M_{T+k}^{(f)}\bm x_{T+k-1}^{(f)},
& r_{T+k,n}^{C}&:=y_{T+k,n}-(\bm h_{T+k,n}^{(f)})^\T\bm x_{T+k}^{(f)}.
\label{eq:joint_residuals}
\end{align}
Then, up to additive constants independent of unknowns:
\begin{align}
\log p_A
=\;&
\log p(\bm\alpha_0)+\log p(\Theta_A)
-\frac12\sum_{t=1}^{T}\!\left[
\log|\tilde{\bm W}_t|+(\bm e_t^A)^\T \tilde{\bm W}_t^{-1}\bm e_t^A
\right] \nonumber\\
&-\frac12\sum_{t=1}^{T}\!\left[
\log \sigma_0^2+\frac{(r_t^0)^2}{\sigma_0^2}
\right],
\label{eq:log_joint_A}
\end{align}
\begin{align}
\log p_B
=\;&
\log p_A
+\sum_{j=1}^{J}\log p(\bm\delta_0^j)
-\frac12\sum_{j=1}^{J}\sum_{t=1}^{T}\!\left[
\log|\bm W_t^{\delta^j}|+(\bm e_t^{\delta,j})^\T(\bm W_t^{\delta^j})^{-1}\bm e_t^{\delta,j}
\right] \nonumber\\
&-\frac12\sum_{j=1}^{J}\sum_{t=1}^{T}\!\left[
\log \sigma_j^2+\frac{(r_t^j)^2}{\sigma_j^2}
\right],
\label{eq:log_joint_B}
\end{align}
\begin{align}
\log p_C
=\;&
\log p(\Phi_C)
-\frac12\sum_{k=1}^{K_{\max}}\!\left[
\log|\bm W_{T+k}^{(f)}|+(\bm e_{T+k}^{(f)})^\T(\bm W_{T+k}^{(f)})^{-1}\bm e_{T+k}^{(f)}
\right] \nonumber\\
&-\frac12\sum_{k=1}^{K_{\max}}\sum_{n=1}^{n_k}\!\left[
\log \sigma^2_{\mathrm{src}(k,n)}+\frac{(r_{T+k,n}^{C})^2}{\sigma^2_{\mathrm{src}(k,n)}}
\right].
\label{eq:log_joint_C}
\end{align}
Equation \eqref{eq:log_joint_C} is the conditional forecast-block log-joint given
$\bm x_T^{(f)}$ and $\sigma_{1:J_f}^2$.
If standalone Model C analysis is used, add the Gaussian-anchor term $\log p(\bm x_T^{(f)})$
and the observation-variance prior terms $\sum_{j=1}^{J_f}\log p(\sigma_j^2)$.
In ragged-horizon Model C, both the dimension of $\bm x_{T+k}^{(f)}$ and the dimension of
$\bm W_{T+k}^{(f)}$ depend on the chosen reduced or augmented forecast specification.
Equations \eqref{eq:log_joint_A}--\eqref{eq:log_joint_C} are the direct objectives used to
compute sufficient statistics for both Gibbs updates and CAVI blocks.
